mutate 新增列

Code 
flights_sml <- select(
  flights,
  year:day,
  ends_with("delay"),
  distance,
  air_time
) |>
  mutate(
    gain = arr_delay - dep_delay,
    speed = distance / air_time * 60,
    .keep = 'used'
  )
