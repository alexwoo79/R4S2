提取方式 语法示例 结果（基于矩阵 A） 说明
单下标（列优先） A[1] 1 第 1 列第 1 行
A[4] 2 第 2 列第 1 行（列优先遍历）

A[, -2] 第 1 列的所有行 -2 表示排除第 2 列
条件筛选（逻辑向量） A[A > 0] 1 4 2 1 提取所有大于 0 的元素

    if (!is.numeric(x)) {

    if (min(x) <= 0) {
      stop("Input must be greater than zero")
    }
  )
  exp(mean(log(x)))

fibonacci <- function (n) {
a <- 1
b <- 0
while (n > 0) {
swap <- a
a <- a + b
b <- swap
n <- n - 1
}
b
}
